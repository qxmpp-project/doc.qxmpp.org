130 template<typename T>
131 QFuture<QXmppPubSubManager::ItemResult<T>> QXmppPubSubManager::requestItem(const QString &jid,
132  const QString &nodeName,
133  const QString &itemId)
134 {
135  using namespace QXmpp::Private;
136  using Error = QXmppStanza::Error;
137  return chainIq(client()->sendIq(requestItemsIq(jid, nodeName, { itemId })), this,
138  [](QXmppPubSubIq<T> &&iq) -> ItemResult<T> {
139  if (!iq.items().isEmpty()) {
140  return iq.items().constFirst();
141  }
142  return Error(Error::Cancel, Error::ItemNotFound, QStringLiteral("No such item has been found."));
143  });
144 }
