149 template<typename T>
150 QFuture<QXmppPubSubManager::ItemResult<T>> QXmppPubSubManager::requestItem(const QString &jid,
151  const QString &nodeName,
152  const QString &itemId)
153 {
154  using namespace QXmpp::Private;
155  using Error = QXmppStanza::Error;
156  return chainIq(client()->sendIq(requestItemsIq(jid, nodeName, { itemId })), this,
157  [](QXmppPubSubIq<T> &&iq) -> ItemResult<T> {
158  if (!iq.items().isEmpty()) {
159  return iq.items().constFirst();
160  }
161  return Error(Error::Cancel, Error::ItemNotFound, QStringLiteral("No such item has been found."));
162  });
163 }
