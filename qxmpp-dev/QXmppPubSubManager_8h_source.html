139 template<typename T>
140 QFuture<QXmppPubSubManager::ItemResult<T>> QXmppPubSubManager::requestItem(const QString &jid,
141  const QString &nodeName,
142  const QString &itemId)
143 {
144  using namespace QXmpp::Private;
145  using Error = QXmppStanza::Error;
146  return chainIq(client()->sendIq(requestItemsIq(jid, nodeName, { itemId })), this,
147  [](QXmppPubSubIq<T> &&iq) -> ItemResult<T> {
148  if (!iq.items().isEmpty()) {
149  return iq.items().constFirst();
150  }
151  return Error(Error::Cancel, Error::ItemNotFound, QStringLiteral("No such item has been found."));
152  });
153 }
