173 template<typename T>
174 QFuture<QXmppPubSubManager::ItemResult<T>> QXmppPubSubManager::requestItem(const QString &jid,
175  const QString &nodeName,
176  const QString &itemId)
177 {
178  using namespace QXmpp::Private;
179  using Error = QXmppStanza::Error;
180  return chainIq(client()->sendIq(requestItemsIq(jid, nodeName, { itemId })), this,
181  [](QXmppPubSubIq<T> &&iq) -> ItemResult<T> {
182  if (!iq.items().isEmpty()) {
183  return iq.items().constFirst();
184  }
185  return Error(Error::Cancel, Error::ItemNotFound, QStringLiteral("No such item has been found."));
186  });
187 }
