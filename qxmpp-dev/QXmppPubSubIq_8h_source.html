163 template<typename T>
164 void QXmppPubSubIq<T>::parseItems(const QDomElement &queryElement)
165 {
166  for (auto childElement = queryElement.firstChildElement(QStringLiteral("item"));
167  !childElement.isNull();
168  childElement = childElement.nextSiblingElement(QStringLiteral("item"))) {
169  T item;
170  item.parse(childElement);
171  m_items << std::move(item);
172  }
173 }
